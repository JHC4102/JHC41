定義抓取一頁資料的函數 (class="ustyle_heading_H3",span="sourceTitleName",抓取當前網頁中所有的a)


def extract_page_data():
    time.sleep(1)
    
    # 同時抓取標題與來源名稱元素
    items = driver.find_elements(By.CSS_SELECTOR, ".ustyle_heading_H3")
    sources = driver.find_elements(By.CSS_SELECTOR, "span.sourceTitleName")

    for i in range(min(len(items), len(sources))):
        try:
            # 抓取標題
            title_element = items[i].find_element(By.TAG_NAME, "a")
            title = title_element.text.strip()
        except:
            title = "無標題"
            
        try:
            # 抓取來源
            source = sources[i].text.strip()
        except:
            source = "無來源"
            
            # 加入資料
        data_list.append({
            "標題": title,
            "來源": source
        })
        print(f"標題：{title}")
        print(f"來源：{source}")
        print("-" * 50)

開啟chrome進入網站並搜尋【梅花鹿】


options = webdriver.ChromeOptions()
options.add_argument("--disable-blink-features=AutomationControlled")
driver = webdriver.Chrome(
    service=webdriver.chrome.service.Service(ChromeDriverManager().install()),
    options=options
)
# 啟動瀏覽器
driver.get("https://www.airitilibrary.com/")
driver.maximize_window()
# 搜尋關鍵字
search_box = driver.find_element(By.ID, "_Search_檢索列") 
search_box.send_keys("梅花鹿")
search_box.send_keys(Keys.ENTER)

抓取多頁及儲存資料


# 存放所有資料的列表
data_list = []
# 迴圈抓多頁(此函式定義抓取三頁)
count=0
while count<3:
    time.sleep(2)
    extract_page_data()
    try:
        next_button = driver.find_element(By.CSS_SELECTOR,"li.page.PagedList-skipToNext a[rel='next']") 
        if next_button.is_enabled():
            next_button.click()
        else:
            break  # 已經是最後一頁
    except (NoSuchElementException, ElementNotInteractableException):
        break
    count +=1
driver.quit()
